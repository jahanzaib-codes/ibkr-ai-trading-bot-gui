//+------------------------------------------------------------------+
void CControlsDialog::ColorBackground(const color clr)
  {
   string prefix=Name();
   int total=ControlsTotal();
   for(int i=0; i<total; i++)
     {
      CWnd*obj=Control(i);
      string name=obj.Name();
      //---
      if(name==prefix+"Client")
        {
         CWndClient *wndclient=(CWndClient*) obj;
         wndclient.ColorBackground(clr);
         wndclient.ColorBorder(clr);
         ChartRedraw();
         return;
        }
     }
//---
  }
//+------------------------------------------------------------------+
//|                                                                  |
//+------------------------------------------------------------------+
void CControlsDialog::ColorCaption(const color clr)
  {
   string prefix=Name();
   int total=ControlsTotal();
// color clr=clrBlack;
   for(int i=0; i<total; i++)
     {
      CWnd*obj=Control(i);
      string name=obj.Name();
      //---
      if(name==prefix+"Caption")
        {
         CEdit *edit=(CEdit*) obj;
         edit.ColorBackground(clr);
         edit.Color(clrWhite);
        }
     }
//--- Return the color
// return clr;
  }

bool CControlsDialog::CreateLabel0(void)  // Trend Info Symbol Lable
  {
//--- coordinates
   int x1 = dashWidth/4 + 20;
   int y1 = dashHeight - dashHeight + 23;
   int x2 = x1 + 20;
   int y2 = y1 + 20;
//--- create
   if(!m_label0.Create(m_chart_id,m_name+"volume",m_subwin,x1,y1,x2,y2))
      return(false);
   if(!m_label0.Text(string(iVolume(Symbol(),PERIOD_CURRENT,1))))
      return(false);
   if(!m_label0.Color(clrBlack))
      return(false);
   if(!m_label0.Font("Calibri"))
      return(false);
   if(!m_label0.FontSize(fontSize))
      return(false);
   if(!Add(m_label0))
      return(false);
//--- succeed
   return(true);
  }

bool CControlsDialog::CreatePanel(void)
  {
//--- coordinates
   int x1=0;
   int y1=0;
   int x2=dashWidth;
   int y2=dashHeight;
//--- create
   if(!backgroundPanel.Create(0,ExtDialog.Name()+"backPanel",m_subwin,x1,y1,x2,y2))
      return(false);
   if(!backgroundPanel.ColorBackground(clrWhiteSmoke))
      return(false);
   if(!backgroundPanel.ColorBorder(00128))
      return(false);
   if(!ExtDialog.Add(backgroundPanel))
      return(false);
   ExtDialog.Minimized(false);


   backgroundPanel.Alignment(WND_ALIGN_CLIENT,0,0,0,0);
//--- succeed
   return(true);
  }

ulong placeBuyTrade(string tradeComment,double lot)
  {

   double price  = SymbolInfoDouble(Symbol(),SYMBOL_ASK);
   double buySl = 0,buyTp = 0;

   if(trade.Buy(lot,Symbol(),price,buySl,buyTp,tradeComment))
     {
      Print(" Buy Trade Placed: ",trade.ResultOrder());
      return trade.ResultOrder();
     }
   else
     {
      Print(" Error in placing Buy Trade : ",GetLastError());
     }
   return -1;
  }
